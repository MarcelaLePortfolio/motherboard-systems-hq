    63	      data = (maybe.payload !== undefined ? maybe.payload : maybe);
    64	    }

   122	export function deriveHighestSeverity(
   123	  signals: GovernanceAdvisorySignal[]
   124	): GovernanceAdvisoryReport["highest_severity"] {
   125	
   126	  if (signals.length === 0) {
   127	    return "none";
   128	  }
   129	
   130	  const order = [
   131	    "info",
   132	    "notice",
   133	    "warning",
   134	    "risk",
   135	    "critical"
   136	  ];
   137	
   138	  let highest = 0;
   139	
   140	  for (const s of signals) {
   141	    const index = order.indexOf(s.severity);
   142	
   143	    if (index > highest) {
   144	      highest = index;
   145	    }
   146	  }
   147	
   148	  return order[highest] as GovernanceAdvisoryReport["highest_severity"];
   149	}
